Python 
from titanforge import TitanClient

# Initialiseer de client
client = TitanClient(api_key="tf_live_xxxxxx")

# Stuur een bericht naar de AI
response = client.chat.complete(
    model="titan-pro-2025",
    messages=[
        {
            "role": "system",
            "content": "Je bent een behulpzame assistent."
        },
        {
            "role": "user",
            "content": "Schrijf een Python functie voor sorteren"
        }
    ],
    max_tokens=1024,
    temperature=0.7
)

print(response.choices[0].message.content)
# → "def bubble_sort(arr): ..."
